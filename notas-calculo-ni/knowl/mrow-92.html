\begin{align*} f'(x) =\amp~ \lim_{h\rightarrow 0} \frac{\frac{1}{(x+h)^n}-\frac{1}{x^n}}{h}\\ =\amp~\lim_{h\rightarrow 0} -\frac{(x+h)^n-x^n}{h}\cdot\frac{1}{(x+h)^nx^n}\\ =\amp~-\lim_{h\rightarrow 0} \frac{(x+h)^n-x^n}{h}\cdot\lim_{h\rightarrow 0} \frac{1}{(x+h)^nx^n}\\ =\amp~ -nx^{n-1}\cdot \frac{1}{x^{2n}}\\ =\amp~ -nx^{-n-1}. \end{align*}
