  4. \(\displaystyle \mbox{cotg}\hspace{0.05cm}\left(\dfrac{-31\pi}{4}\right)\)
  5. \(\displaystyle \sec\dfrac{71\pi}{6}\)
  6. \(\displaystyle \mbox{cotg}\hspace{0.05cm}\left(\dfrac{-91\pi}{3}\right)\)
