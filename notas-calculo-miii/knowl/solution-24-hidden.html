\begin{equation*} \left(1+\frac{1}{x} \right)^x = \left( 1-\frac{1}{1+u} \right)^{-u-1} = \left( 1+\frac{1}{u} \right)^u\cdot \frac{u+1}{u}. \end{equation*}
