\begin{equation*} A_2 = \int_{0}^{1} (x - x^3) \, dx = \left. \frac{x^2}{2} - \frac{x^4}{4} \right|_{0}^{1} \end{equation*}
\begin{equation*} A_2 = \left( \frac{1}{2} - \frac{1}{4} \right) - 0 = \frac{1}{4} \end{equation*}
